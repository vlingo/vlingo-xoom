package io.vlingo.xoomapp.infrastructure;

import io.vlingo.xoomapp.model.*;

public class ClassificationData {

  public final String label;
  public final ClassifierData classifier;

  public static ClassificationData from(final Classification classification) {
    return from(classification.label, classification.classifier);
  }

  public static ClassificationData from(final String label, final Classifier classifier) {
    return new ClassificationData(label, classifier);
  }

  private ClassificationData (final String label, final Classifier classifier) {
    this.label = label;
    this.classifier = classifier != null ? ClassifierData.from(classifier) : null;
  }

}
