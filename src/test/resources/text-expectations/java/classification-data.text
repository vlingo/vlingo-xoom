package io.vlingo.xoomapp.infrastructure;

import io.vlingo.xoomapp.model.Classification;

public class ClassificationData {

  public final String label;
  public final ClassifierData classifier;

  public static ClassificationData from(final Classification classification) {
    final ClassifierData classifier = classification.classifier != null ? ClassifierData.from(classification.classifier) : null;
    return from(classification.label, classifier);
  }

  public static ClassificationData from(final String label, final ClassifierData classifier) {
    return new ClassificationData(label, classifier);
  }

  private ClassificationData (final String label, final ClassifierData classifier) {
    this.label = label;
    this.classifier = classifier;
  }

}
