package io.vlingo.xoomapp.infrastructure;

import io.vlingo.xoomapp.model.*;

public class RankData {

  public final int points;
  public final ClassificationData classification;

  public static RankData from(final Rank rank) {
    return from(rank.points, rank.classification);
  }

  public static RankData from(final int points, final Classification classification) {
    return new RankData(points, classification);
  }

  private RankData (final int points, final Classification classification) {
    this.points = points;
    this.classification = classification != null ? ClassificationData.from(classification) : null;
  }

}
